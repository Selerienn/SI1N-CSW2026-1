Código Fonte

def calcular_f_de_x(a, b, c, x):
    """
    Calcula o valor de f(x) para uma função de segundo grau.
    """
    return a * (x**2) + b * x + c

# Entrada de dados
print("--- Calculadora de f(x) = ax² + bx + c ---")
a = float(input("Digite o valor de a: "))
b = float(input("Digite o valor de b: "))
c = float(input("Digite o valor de c: "))
x = float(input("Digite o valor de x: "))

# Cálculo
resultado = calcular_f_de_x(a, b, c, x)

# Saída
print(f"\nPara a função f(x) = {a}x² + {b}x + {c}")
print(f"O valor de f({x}) é: {resultado}")
